   ```typescript
   import { format, startOfDay, isSameDay, parseISO } from 'date-fns';
   import type { PostureRecord } from '../../data/types';

   /**
    * Extract unique dates (YYYY-MM-DD) present in the dataset.
    * Returns sorted array of ISO date strings.
    */
   export function extractAvailableDates(records: PostureRecord[]): string[] {
     const dates = new Set<string>();
     for (const r of records) {
       dates.add(format(new Date(r.time), 'yyyy-MM-dd'));
     }
     return Array.from(dates).sort();
   }

   /**
    * Filter records to only those on the specified date.
    */
   export function extractDayRecords(records: PostureRecord[], dateStr: string): PostureRecord[] {
     const target = parseISO(dateStr);
     return records.filter(r => isSameDay(new Date(r.time), target));
   }

   /**
    * Convert timestamp to minutes since midnight (0-1439).
    * Used to normalize X-axis when comparing days (per D-06).
    */
   export function normalizeToMinuteOfDay(timeMs: number): number {
     const d = new Date(timeMs);
     return d.getHours() * 60 + d.getMinutes();
   }
   ```

   - Need to provide `timeToX` and `deltaYToY` functions. For Recharts, this requires accessing the chart's internal scale. Two approaches:
     a. Use Recharts `useChart` hook (if available in v3.8) to get xScale/yScale
     b. Or: compute approximate positions from the ResponsiveContainer width and the data domain

   **Recommended approach for Recharts:** Wrap the chart in a container div and position AnnotationLayer absolutely over it. Calculate X/Y from data domain + container dimensions:
   ```typescript
   const containerRef = useRef<HTMLDivElement>(null);

   const timeToX = useCallback((time: number) => {
     if (!containerRef.current) return 0;
     const rect = containerRef.current.getBoundingClientRect();
     const chartWidth = rect.width - marginLeft - marginRight;
     const [domainMin, domainMax] = visibleDomain ?? [
       Math.min(...data.map(d => d.time)),
       Math.max(...data.map(d => d.time)),
     ];
     return marginLeft + ((time - domainMin) / (domainMax - domainMin)) * chartWidth;
   }, [data, visibleDomain]);

   const deltaYToY = useCallback((deltaY: number) => {
     if (!containerRef.current) return 0;
     const rect = containerRef.current.getBoundingClientRect();
     const chartHeight = rect.height - marginTop - marginBottom;
     const yValues = data.filter(d => d.deltaY !== null).map(d => d.deltaY!);
     const yMin = Math.min(...yValues, -thresholdPx);
     const yMax = Math.max(...yValues, thresholdPx);
     const padding = (yMax - yMin) * 0.1;
     const fullMin = yMin - padding;
     const fullMax = yMax + padding;
     return marginTop + ((fullMax - deltaY) / (fullMax - fullMin)) * chartHeight;
   }, [data, thresholdPx]);
   ```
